PHP HTML Result 
{content 'cell'} 
  <td {if $cell EQ 23} bgcolor="red" {/if} >
    {$cell}
  </td>
{/content}
 
<?php

  for ($row = 1; $row <= 3; $row++)
    for ($col = 1; $col <= 5; $col++)     
      $rows [$row] ['cols'] [$col] ['cell'] = ($row*10) + $col;

?>
 
<table border='1'>
  {rows}
    <tr>
      {cols}
        {cell}  
      {/cols}
    </tr>
  {/rows}
</table>

<?php

  $row    = [1,2,3];
  $column = [1,2,3,4,5];

?>
 
<table border='1'>
  {row}
    <tr>
      {column}
        {cell ($row*10) + $column} 
      {/column}
    </tr>
  {/row}
</table>
 
{data 'row'   } [1,2,3]     {/data}

<table border='1'>
  {row}
    <tr>

<table border='1'>
  {sequence '10..30', increment=10}
    <tr>
      {sequence '1..5'}
        {cell $-2 + $-1}
      {/sequence}
    </tr>
  {/sequence}
</table>
 
<table border='1'>
  {sequence '10..30', increment=10, name='row'}
    <tr>
      {sequence '1..5', name='column'}
        {cell $row + $column}
      {/sequence}
    </tr>
  {/sequence}
</table>
 
<table border='1'>
  {set $row = 10}
  {while $row LE 30}
    <tr>
      {set $col = 1}
      {while $col LE 5}
        {cell $row + $col}
        {increment $col}
      {/while}
    </tr>
    {set $row = $row + 10}
  {/while}
</table>
